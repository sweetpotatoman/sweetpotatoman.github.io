gacp: git-add-commit-push

# 一个简单的 gacp (Mac 终端)

  • 在 Public 目录下创建一个 gacp.sh 脚本并添加以下内容
#!/bin/sh

set -x  # Output executed commands
set -e  # Make script fail as soon as one command fails

read MESSAGE

git add -A
git commit -m "$MESSAGE"
git push origin master
  • 修改 .zshrc 文件，添加 alias
alias gacp='sh ~/Public/gacp.sh'
  • 在某个项目下使用 gacp
~/Work> gacp
